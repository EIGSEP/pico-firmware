cmake_minimum_required(VERSION 3.13)

# Include pico_sdk_import.cmake from switches app
include(apps/switches/pico_sdk_import.cmake)

# Set project name and language  
project(test_dip C CXX ASM)

# Set C/C++ standards
set(CMAKE_C_STANDARD 11)
set(CMAKE_CXX_STANDARD 17)

# Initialize the Pico SDK
pico_sdk_init()

# Test executable for DIP switch testing
add_executable(test_dip
    src/test_dip.c
)

# Define TEST_DIP_STANDALONE for standalone compilation
target_compile_definitions(test_dip PRIVATE TEST_DIP_STANDALONE=1)

# Include directories
target_include_directories(test_dip PRIVATE
    ${CMAKE_CURRENT_LIST_DIR}/src
)

# Link libraries
target_link_libraries(test_dip 
    pico_stdlib
    hardware_gpio
)

# Enable USB output, disable UART output
pico_enable_stdio_usb(test_dip 1)
pico_enable_stdio_uart(test_dip 0)

# Create map/bin/hex/uf2 files
pico_add_extra_outputs(test_dip)
